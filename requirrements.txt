```python
"""
Pdf chatbot
"""
import spacy

def load_model(model_name):
    """
    Load the specified spaCy model.

    Args:
        model_name (str): The name of the spaCy model to load.

    Returns:
        Language: The loaded spaCy model.
    """
    nlp = spacy.load(model_name)
    return nlp

def process_text(nlp, text):
    """
    Process the input text using the loaded spaCy model.

    Args:
        nlp (Language): The loaded spaCy model.
        text (str): The text to process.

    Returns:
        Doc: The processed spaCy Doc object.
    """
    doc = nlp(text)
    return doc
```
